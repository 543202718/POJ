# 图算法
## POJ1201
差分约束系统。x[i]表示在区间[0,i]中选择的数的个数。则有：
$$
x[a_i-1]\le x[b_i]+(-c_i) \\
x[i+1]\le x[i]+1 \\
x[i] \le x[i+1] + 0 
$$

上述不等式可以等价于$b_i$到$a_i-1$权重为$-c_i$的边，以及i到i+1权重为1的边，以及i+1到i权重为0的边。


```cpp
#include <cstdio>
#define MAXV 50000
using namespace std;

struct edge {
	int u, v, w;
};

edge edges[MAXV * 4];
int dist[MAXV + 10];
int m;
void bellman_ford(int left, int right) {
	for (int i = left; i <= right; i++) {
		dist[i] = -1;
	}
	dist[left] = 0;
	for (int i = left; i <= right; i++) {
		bool flag = false;
		for (int k = 0; k < m; k++) {
			int u = edges[k].u, v = edges[k].v, w = edges[k].w;
			if (dist[u] != -1 && dist[v] < dist[u] + w) {
				dist[v] = dist[u] + w;
				flag = true;
			}
		}
		for (int j = left; j <= right - 1; j++) {
			if (dist[j] != -1 && dist[j] > dist[j + 1]) {
				dist[j + 1] = dist[j];
				flag = 1;
			}
		}
		for (int j = right; j >= left + 1; j--) {
            //这里的顺序很重要，如果反序，就会超时
			if (dist[j] != -1 && dist[j] - 1 > dist[j - 1]) {
				dist[j - 1] = dist[j] - 1;
				flag = 1;
			}
		}
		if (!flag) break;
	}
}
int main() {
	int n, left = MAXV, right = 0;
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		int a, b, c;
		scanf("%d%d%d", &a, &b, &c);
		edges[i] = { a,b + 1,c };
		left = left < a ? left : a;
		right = right > b ? right : b;
	}
	right++;
	m = n;
	bellman_ford(left, right);
	printf("%d\n", dist[right] - dist[left]);
	return 0;
}
```

## POJ2983
差分约束系统。s[i]表示第i个防御站的位置，则$P\ A\ B\ X$意味着$S[A]-S[B]=X$，而$V\ A\ B$意味着$S[A] \ge S[B]+1$。将前一个式子转化为两个不等式$S[A]-S[B] \ge X$和$S[A]-S[B] \le X$，对应于A到B权为-X的边和B到A权为X的边。后一个式子对应于A到B权为-1的边。为了保证图的连通性，增加一个超级源点0，它到所有点都有一条权值为0的边。

原始问题等价于图中是否有权值为负的环。采用SPFA求解，如果一个点被加入队列超过n次（n是顶点数目），那么图中有负权环路。

**cin和cout的效率远远低于scanf和printf，下面的代码使用cin和cout就会超时。**



```cpp
#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
#include<cmath>
#include<queue>
#define INF 999999
using namespace std;

struct edge {
	int v, w, next;
};

edge edges[300000], node[1010];
queue<int> q;
int dist[1010];
bool flag[1010];
int times[1010];
int n, m, counter;

void add(int u, int v, int w) {
	edges[counter] = { v,w,node[u].next };
	node[u].next = counter;
	counter++;
}
bool spfa() {
	for (int i = 1; i <= n; i++) {
		dist[i] = INF;
		flag[i] = false;
		times[i] = 0;
	}
	q = queue<int>();//clear the queue
	dist[0] = 0;
	q.push(0);
	flag[0] = true;
	times[0] = 1;
	while (!q.empty()) {
		int top = q.front();
		q.pop();
		flag[top] = false;
		for (int i = node[top].next; i != -1; i = edges[i].next) {
			int v = edges[i].v, w = edges[i].w;
			if (dist[v] > dist[top] + w) {
				dist[v] = dist[top] + w;
				if (!flag[v]) {
					q.push(v);
					flag[v] = true;
					times[v]++;
					if (times[v] > n) {
						return false;
					}
				}
			}
		}
	}
	return true;
}
int main() {
	int m, a, b, c;
	char ch[2];
	while (cin >> n >> m){
		for (int i = 0; i <= n; i++) node[i].next = -1;
		counter = 0;
		while (m--)
		{
			scanf("%s", &ch);
			if (ch[0] == 'P')
			{
				scanf("%d%d%d", &a, &b, &c);
				add(a, b, -c);
				add(b, a, c);
			}
			else
			{
				scanf("%d%d", &a, &b);
				add(a, b, -1);
			}
		}
		for (int i = 1; i <= n; i++)
			add(0, i, 0);
		if (!spfa()) puts("Unreliable");
		else puts("Reliable");
	}
	return 0;
}
```

## POJ2516
最小费用最大流。将k种货物拆开成k个图，就是一个多个源和多个汇的网络流问题。增加超级源和超级汇，超级源到所有供应点有一条边，流量为供应点的供应量，单价为0；所有商店到超级汇有一条边，流量为商店的需求量，单价为0；供应点和商店间有边，流量为无穷，单价从输入中读取，再构建流量为0单价为相反数的反向边。

使用SPFA寻找单价的最短路，再修改网络（增广边），继续SPFA直到找不到从源到汇的路径。

当供应量小于需求量时，找不到合适的解。
```cpp
#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
#include<cmath>
#include<queue>
#define INF 999999
using namespace std;
queue<int> q;
int dist[300];
bool flag[300];
int n, m, k, counter;
int source[300][300], sink[300][300];
int a[300][300];
int flow[300][300], cost[300][300];

int pre[300];

bool spfa() {
	for (int i = 0; i <= m + n + 1; i++) {
		dist[i] = INF;
		flag[i] = false;
		pre[i] = -1;
	}
	q = queue<int>();//clear the queue
	dist[0] = 0;
	q.push(0);
	flag[0] = true;
	while (!q.empty()) {
		int top = q.front();
		q.pop();
		flag[top] = false;
		for (int i = 0; i <= m + n + 1; i++) {
			if (flow[top][i] > 0 && dist[i] > dist[top] + cost[top][i]) {
				dist[i] = dist[top] + cost[top][i];
				pre[i] = top;
				if (!flag[i]) {
					q.push(i);
					flag[i] = true;
				}
			}
		}
	}
	return (pre[m + n + 1] != -1);
}

int minCost(int k) {
	memset(flow, 0, sizeof(flow));
	memset(cost, INF, sizeof(cost));
	//0是超级源，1..m是供应点,m+1..m+n是商店，m+n+1是超级汇
	for (int i = 1; i <= n; i++) {
		flow[m + i][m + n + 1] = sink[i][k];
		cost[m + i][m + n + 1] = 0;
	}//增加所有商店到超级汇的边，流量为商店的需求量，单价为0
	for (int i = 1; i <= m; i++) {
		flow[0][i] = source[i][k];
		cost[0][i] = 0;
	}//增加超级源到所有供应点的边，流量为供应点的供应量，单价为0
	for (int i = 1; i <= n; i++) {
		for (int j = 1; j <= m; j++) {
			flow[j][i + m] = INF;
			cost[j][i + m] = a[i][j];
			flow[i + m][j] = 0;
			cost[i + m][j] = -a[i][j];
		}
	}//增加所有供应点到商店的边，流量为无穷，单价为读入的值;以及反向边，流量为0，单价为相反数
	int ret = 0;
	while (spfa()) {
		int minflow = INF;
		for (int i = m + n + 1; i != 0; i = pre[i]) {
			if (minflow > flow[pre[i]][i]) minflow = flow[pre[i]][i];
		}
		for (int i = m + n + 1; i != 0; i = pre[i]) {
			flow[pre[i]][i] -= minflow;
			flow[i][pre[i]] += minflow;
			ret += (cost[pre[i]][i] * minflow);
		}
	}
	return ret;
}
int main() {
	while (true) {
		scanf("%d%d%d", &n, &m, &k);
		if (n + m + k == 0) break;
		for (int i = 1; i <= n; i++) {
			for (int j = 0; j < k; j++) {
				scanf("%d", &sink[i][j]);
			}
		}
		for (int i = 1; i <= m; i++) {
			for (int j = 0; j < k; j++) {
				scanf("%d", &source[i][j]);
			}
		}
		int ans = 0;
		for (int i = 0; i < k; i++) {
			for (int p = 1; p <= n; p++) {
				for (int q = 1; q <= m; q++) {
					scanf("%d", &a[p][q]);
				}
			}
			//读入价格矩阵
			if (ans == -1) continue;
			int sum_src = 0, sum_sink = 0;
			for (int j = 1; j <= m; j++) {
				sum_src += source[j][i];
			}
			for (int j = 1; j <= n; j++) {
				sum_sink += sink[j][i];
			}
			if (sum_src < sum_sink) {
				ans = -1;
				continue;
			}
			//计算并判断是否无法满足要求
			ans += minCost(i);
		}
		printf("%d\n", ans);
	}
	return 0;
}
```

## POJ2195
最小费用最大流问题。将小人视作源，房子视作汇，容量为1，单价为曼哈顿距离，再增加超级源和超级汇即可。由于小人和房子一样多，不需要判定能否满足要求。

```cpp
#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
#include<cmath>
#include<queue>
#define INF 999999

using namespace std;
queue<int> q;
int dist[300];
bool flag[300];
int n, m, k, counter;
int source[300][2], sink[300][2];
int a[300][300];
int flow[300][300], cost[300][300];
int cnt1, cnt2;
int pre[300];

bool spfa() {
	int n = cnt1;
	for (int i = 0; i <= n + n + 1; i++) {
		dist[i] = INF;
		flag[i] = false;
		pre[i] = -1;
	}
	q = queue<int>();//clear the queue
	dist[0] = 0;
	q.push(0);
	flag[0] = true;
	while (!q.empty()) {
		int top = q.front();
		q.pop();
		flag[top] = false;
		for (int i = 0; i <= n + n + 1; i++) {
			if (flow[top][i] > 0 && dist[i] > dist[top] + cost[top][i]) {
				dist[i] = dist[top] + cost[top][i];
				pre[i] = top;
				if (!flag[i]) {
					q.push(i);
					flag[i] = true;
				}
			}
		}
	}
	return (pre[n + n + 1] != -1);
}

int minCost() {
	int n = cnt1;
	memset(flow, 0, sizeof(flow));
	memset(cost, INF, sizeof(cost));
	//0是超级源，1..n是小人,n+1..2n是房子，2n+1是超级汇
	for (int i = 1; i <= n; i++) {
		flow[n + i][n + n + 1] = 1;
		cost[n + i][n + n + 1] = 0;
	}//增加所有房子到超级汇的边，流量为1，单价为0
	for (int i = 1; i <= n; i++) {
		flow[0][i] = 1;
		cost[0][i] = 0;
	}//增加超级源到所有小人的边，流量为1，单价为0
	for (int i = 1; i <= n; i++) {
		for (int j = 1; j <= n; j++) {
			flow[i][j + n] = 1;
			cost[i][j + n] = abs(source[i][0] - sink[j][0]) + abs(source[i][1] - sink[j][1]);//曼哈顿距离
			flow[n + j][i] = 0;
			cost[n + j][i] = -cost[i][j + n];
		}
	}//增加所有小人到房子的边，流量为1，单价为曼哈顿距离;以及反向边，流量为0，单价为相反数
	int ret = 0;
	while (spfa()) {
		int minflow = INF;
		for (int i = n + n + 1; i != 0; i = pre[i]) {
			if (minflow > flow[pre[i]][i]) minflow = flow[pre[i]][i];
		}
		for (int i = n + n + 1; i != 0; i = pre[i]) {
			flow[pre[i]][i] -= minflow;
			flow[i][pre[i]] += minflow;
			ret += (cost[pre[i]][i] * minflow);
		}
	}
	return ret;
}
int main() {
	char c;
	while (true) {
		scanf("%d%d", &n, &m);
		if (n + m == 0) break;
		cnt1 = cnt2 = 0;
		for (int i = 1; i <= n; i++) {
			for (int j = 1; j <= m; j++) {
				scanf(" %c", &c);
				if (c == 'H') {
					cnt2++;
					sink[cnt2][0] = i;
					sink[cnt2][1] = j;
				}
				else if (c == 'm') {
					cnt1++;
					source[cnt1][0] = i;
					source[cnt1][1] = j;
				}
			}
		}
		printf("%d\n", minCost());
	}
	return 0;
}
```

## POJ2186
这道题将牛视作节点，将崇拜关系视作边，某些点从其他所有点都可达，求这种点的数目。

记从其他所有点都可达的点满足性质A。容易证明，如果一个点满足性质A，它所在的强连通分量中的所有点都满足性质A；如果两个点都满足性质A，那么它们一定在同一个强连通分量上。

具体做法是，使用Tarjan算法求强连通分量，将每一个强连通分量视作一个点，那么原图就被转化为一个有向无环图。在有向无环图中，如果出度为0的点有两个及以上，那么就找不到这种点（显然，一个出度为0的点无法到达另一个出度为0的点）；如果只有一个，那么这个点对应的强连通分量的大小就是所求的数目。


```cpp
#include<cstdio>
#include<cstdlib>
#include<stack>
#include<algorithm>
using namespace std;

struct edge {
	int v;
	edge *next;
};

edge a[10010];
int m, n;
stack<int> s;

int low[10010], dfn[10010], belong[10010];
bool flag[10010];//标记节点是否在栈中
int ts = 0, block = 0;
int cnt[10010];
void tarjan(int x) {
	ts++;
	s.push(x);
	flag[x] = true;
	low[x] = ts;
	dfn[x] = ts;
	edge* temp = a[x].next;
	while (temp != NULL) {
		if (dfn[temp->v] == 0) {//该节点从未被访问过
			tarjan(temp->v);
			low[x] = min(low[x], low[temp->v]);
		}
		else if (flag[temp->v]) {//该节点被访问过且在栈中
			low[x] = min(low[x], low[temp->v]);
		}
		temp = temp->next;
	}
	int v;
	if (low[x] == dfn[x]) {
		block++;
		do {
			v = s.top();
			s.pop();//出栈
			belong[v] = block;
			flag[v] = false;
		} while (v != x);
	}
}

int cal() {
	for (int i = 1; i <= n; i++) {
		if (cnt[belong[i]] != -1) {
			cnt[belong[i]]++;//统计该强连通分量的节点数目
			for (edge* temp = a[i].next; temp != NULL; temp = temp->next) {
				if (belong[temp->v] != belong[i]) {//表明该强连通分量的出度不为0
					cnt[belong[i]] = -1;
					break;
				}
			}
		}
	}
	int ans = -1;
	for (int i = 1; i <= block; i++) {
		if (ans == -1 && cnt[i] != -1)	ans = cnt[i];
		else if (cnt[i] != -1) ans = 0;
	}
	return ans;
}


int main() {
	scanf("%d%d", &n, &m);
	for (int i = 1; i <= n; i++) {
		a[i].next = NULL;
	}//邻接表初始化
	for (int i = 0; i < m; i++) {
		int u, v;
		scanf("%d%d", &u, &v);
		edge* temp = (edge*)malloc(sizeof(edge));
		temp->v = v;
		temp->next = a[u].next;
		a[u].next = temp;
	}//读取边存入邻接表
	for (int i = 1; i <= n; i++) {
		if (dfn[i] == 0) tarjan(i);
	}
	printf("%d\n", cal());
	return 0;
}
```
## POJ3352
这道题目实际问的是，对一个无向图，至少需要添加多少条边才能使其成为边双联通图。

使用Tarjan算法求双联通分量，缩点，可以得到一棵树。实际上，只要将这一棵树修改为一个双联通图即可。假设树中有n个叶子结点，两两相连，再加$(n+1)/2$条边即可。

```cpp
#include<cstdio>
#include<cstdlib>
#include<stack>
#include<algorithm>
using namespace std;

struct edge {
	int v;
	edge *next;
};

edge a[1010];
int m, n;
stack<int> s;

int low[1010], dfn[1010], belong[1010];
bool flag[1010];//标记节点是否在栈中
int ts = 0, block = 0;
int cnt[1010];
void tarjan(int x, int father) {
	ts++;
	s.push(x);
	flag[x] = true;
	low[x] = ts;
	dfn[x] = ts;
	for (edge* temp = a[x].next; temp != NULL; temp = temp->next) {
		if (temp->v == father) continue;//不处理父结点
		if (dfn[temp->v] == 0) {//该节点从未被访问过
			tarjan(temp->v, x);
			low[x] = min(low[x], low[temp->v]);
		}
		else if (flag[temp->v]) {//该节点被访问过且在栈中
			low[x] = min(low[x], low[temp->v]);
		}
	}
	int v;
	if (low[x] == dfn[x]) {
		block++;
		do {
			v = s.top();
			s.pop();//出栈
			belong[v] = block;
			flag[v] = false;
		} while (v != x);
	}
}

int cal() {
	for (int i = 1; i <= n; i++) {
		for (edge* temp = a[i].next; temp != NULL; temp = temp->next) {
			if (belong[temp->v] != belong[i]) {
				cnt[belong[i]]++;//统计每一个缩点的度
			}
		}
	}
	int ans = 0;
	for (int i = 1; i <= block; i++) {
		if (cnt[i] == 1) ans++;
	}
	return (1+ans)/2;
}


int main() {
	scanf("%d%d", &n, &m);
	for (int i = 1; i <= n; i++) {
		a[i].next = NULL;
	}//邻接表初始化
	for (int i = 0; i < m; i++) {
		int u, v;
		scanf("%d%d", &u, &v);
		edge* temp = (edge*)malloc(sizeof(edge));
		temp->v = v;
		temp->next = a[u].next;
		a[u].next = temp;
		temp = (edge*)malloc(sizeof(edge));
		temp->v = u;
		temp->next = a[v].next;
		a[v].next = temp;
	}//读取边存入邻接表(双向)
	tarjan(1, -1);
	printf("%d\n", cal());
	return 0;
}
```
